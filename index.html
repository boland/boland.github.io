import glob
years = []
for path in glob.glob('names/*.txt'):
    year = pandas.to_datetime(path[9:-4], format='%Y')
    df = pandas.read_csv(path, names=['name','sex','c'])
    df['year'] = year
    years.append(df)
df = pandas.concat(years)
df['date'] = pandas.DatetimeIndex(df.year)
del df['year']
female = df[df['sex']=='F'].groupby(['name', 'date']).sum().unstack(level=0)
male = df[df['sex']=='M'].groupby(['name', 'date']).sum().unstack(level=0)
female.columns = female.columns.droplevel()
male.columns = male.columns.droplevel()

Deep Dreaming with Neural Networks¶

Again, following the first rule of programming, I am using someone else code. This code is from data scientists at google. They have a trained neural network (model) of classified images. This simply means, they showed an algorithm a picture of a duck and told it it was a duck. The network then analyzed the picture and it learns what a duck is.

The point of this example is show you something interesting that you can do with relatively little effort.

Google wanted a way to visualize what was happening in their neural networks. They highlighted features in images that each layer in their network saw. In their blogpost, they showed a few very cool looking images.

Google provided a jupyter notebook to be able to create your own deep dream images¶

I copy and pasted it into my notebook and annotated it below.

# a couple of utility functions for converting to and from Caffe's input image layout
def preprocess(net, img):
    return np.float32(np.rollaxis(img, 2)[::-1]) - net.transformer.mean['data']
def deprocess(net, img):
    return np.dstack((img + net.transformer.mean['data'])[::-1])

Next, a function is defined that highlights the feature the network has seen.

In [15]:
def make_step(net, step_size=1.5, end='inception_4c/output', jitter=32, clip=True):
    '''Basic gradient ascent step.'''

    src = net.blobs['data'] # input image is stored in Net's 'data' blob
    dst = net.blobs[end]

    ox, oy = np.random.randint(-jitter, jitter+1, 2)
    src.data[0] = np.roll(np.roll(src.data[0], ox, -1), oy, -2) # apply jitter shift
            
    net.forward(end=end)
    dst.diff[:] = dst.data  # specify the optimization objective
    net.backward(start=end)
    g = src.diff[0]
    # apply normalized ascent step to the input image
    src.data[:] += step_size/np.abs(g).mean() * g

    src.data[0] = np.roll(np.roll(src.data[0], -ox, -1), -oy, -2) # unshift image
            
    if clip:
        bias = net.transformer.mean['data']
        src.data[:] = np.clip(src.data, -bias, 255-bias)

Here, they provide a function to visualize what the network is seeing.

In [16]:
def deepdream(net, base_img, iter_n=10, octave_n=4, octave_scale=1.4, end='inception_4c/output', clip=True, **step_params):
    # prepare base images for all octaves
    octaves = [preprocess(net, base_img)]
    for i in xrange(octave_n-1):
        octaves.append(nd.zoom(octaves[-1], (1, 1.0/octave_scale,1.0/octave_scale), order=1))
    
    src = net.blobs['data']
    detail = np.zeros_like(octaves[-1]) # allocate image for network-produced details
    for octave, octave_base in enumerate(octaves[::-1]):
        h, w = octave_base.shape[-2:]
        if octave > 0:
            # upscale details from the previous octave
            h1, w1 = detail.shape[-2:]
            detail = nd.zoom(detail, (1, 1.0*h/h1,1.0*w/w1), order=1)

        src.reshape(1,3,h,w) # resize the network's input image size
        src.data[0] = octave_base+detail
        for i in xrange(iter_n):
            make_step(net, end=end, clip=clip, **step_params)
            
            # visualization
            vis = deprocess(net, src.data[0])
            if not clip: # adjust image contrast if clipping is disabled
                vis = vis*(255.0/np.percentile(vis, 99.98))
            showarray(vis)
            print octave, i, end, vis.shape
            clear_output(wait=True)
            
        # extract details produced on the current octave
        detail = src.data[0]-octave_base
    # returning the resulting image
    return deprocess(net, src.data[0])

Our code is just a wrapper that sends an image through several iterations of google's code. In between each step, it rotates and zooms the image by a small amount.

# Define a function to produce multiple frames.
def trip(frame, end='inception_4c/output', frames=10):
    h, w = frame.shape[:2]
    s = 0.05 # scale coefficient
    for i in xrange(frames):
        frame = deepdream(net, frame, end=end)
        
        # Zoom and rotate by just a tad.
        frame = nd.affine_transform(frame, [1-s, 1-s, 1], [h*s/2, w*s/2, 0], order=1)

def tip_calc(price, tax_rate=0.11, tip_rate=0.15):
    tip = price * tip_rate
    tax = price * tax_rate
    print "Meal:\t${:.2f}".format(price)
    print "Tax:\t${:.2f}".format(tax)
    print "Tip:\t${:.2f}".format(tip)
    print "Total:\t${:.2f}".format(price + tip + tax)
